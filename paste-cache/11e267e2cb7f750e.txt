I've examined both the homepage and /my feed. Here's what I found:
Key Observations:

Homepage (/) and /my feed have identical article card structures - both use a[href^="/article/"] links with headings and bias information, but no dates.
Article detail pages have three timestamps:

Published: X hours ago - near the top of the article
Updated: X hours ago - also near the top
Last Updated: X minutes ago - in the "Coverage Details" section



Here's the complete extension code that works on both homepage and /my feed, showing both Published and Last Updated times:
javascript// content.js
(async function() {
  'use strict';

  const CACHE_KEY = 'gn_date_cache';
  const CACHE_DURATION = 5 * 60 * 1000; // 5 minutes

  // Get cached data
  function getCache() {
    try {
      const cached = localStorage.getItem(CACHE_KEY);
      if (cached) {
        const data = JSON.parse(cached);
        // Clear expired entries
        const now = Date.now();
        for (const key in data) {
          if (now - data[key].timestamp > CACHE_DURATION) {
            delete data[key];
          }
        }
        return data;
      }
    } catch (e) {}
    return {};
  }

  function setCache(slug, published, updated) {
    try {
      const cache = getCache();
      cache[slug] = { published, updated, timestamp: Date.now() };
      localStorage.setItem(CACHE_KEY, JSON.stringify(cache));
    } catch (e) {}
  }

  async function fetchArticleDates(slug) {
    const cache = getCache();
    if (cache[slug]) {
      return cache[slug];
    }

    try {
      const response = await fetch(`https://ground.news${slug}`);
      const html = await response.text();
      
      // Parse the HTML to find Published and Updated times
      const parser = new DOMParser();
      const doc = parser.parseFromString(html, 'text/html');
      
      let published = null;
      let updated = null;
      
      // Find elements containing "Published" and "Updated" text
      const allElements = doc.querySelectorAll('*');
      
      for (const el of allElements) {
        const text = el.textContent?.trim();
        
        // Look for "Published" label - its next sibling or child has the time
        if (text === 'Published' && el.children.length > 0) {
          const timeEl = el.querySelector('*');
          if (timeEl) {
            published = timeEl.textContent?.trim();
          }
        }
        
        // Look for "Updated" label
        if (text === 'Updated' && el.children.length > 0) {
          const timeEl = el.querySelector('*');
          if (timeEl) {
            updated = timeEl.textContent?.trim();
          }
        }
      }
      
      // Alternative: Use regex on the raw HTML
      if (!published) {
        const pubMatch = html.match(/>Published<[^>]*>([^<]+)</);
        if (pubMatch) published = pubMatch[1].trim();
      }
      
      if (!updated) {
        const updMatch = html.match(/>Updated<[^>]*>([^<]+)</);
        if (updMatch) updated = updMatch[1].trim();
      }

      if (published || updated) {
        setCache(slug, published, updated);
      }

      return { published, updated };
    } catch (e) {
      console.error('GN Dates: Failed to fetch', slug, e);
      return { published: null, updated: null };
    }
  }

  function createDateElement(published, updated) {
    const container = document.createElement('div');
    container.className = 'gn-dates-extension';
    container.style.cssText = `
      font-size: 11px;
      color: #888;
      margin-top: 4px;
      line-height: 1.4;
      display: flex;
      flex-wrap: wrap;
      gap: 8px;
    `;

    if (published) {
      const pubSpan = document.createElement('span');
      pubSpan.innerHTML = `📅 <strong>Published:</strong> ${published}`;
      container.appendChild(pubSpan);
    }

    if (updated && updated !== published) {
      const updSpan = document.createElement('span');
      updSpan.innerHTML = `🔄 <strong>Updated:</strong> ${updated}`;
      container.appendChild(updSpan);
    }

    return container;
  }

  async function processArticleCards() {
    // Find all article links
    const articleLinks = document.querySelectorAll('a[href^="/article/"]');
    const processedSlugs = new Map(); // slug -> array of link elements

    // Group links by slug (multiple links often point to same article)
    for (const link of articleLinks) {
      const slug = link.getAttribute('href');
      if (!slug || slug === '/article/') continue;
      
      // Skip if already processed
      if (link.closest('.gn-dates-extension') || 
          link.querySelector('.gn-dates-extension')) continue;

      if (!processedSlugs.has(slug)) {
        processedSlugs.set(slug, []);
      }
      processedSlugs.get(slug).push(link);
    }

    // Process each unique article
    for (const [slug, links] of processedSlugs) {
      // Find the "main" link for this article (one with heading inside)
      const mainLink = links.find(l => l.querySelector('h1, h2, h3, h4, h5, h6'));
      if (!mainLink) continue;

      // Check if we've already added dates to this card
      const parent = mainLink.parentElement;
      if (parent?.querySelector('.gn-dates-extension')) continue;

      // Fetch dates
      const { published, updated } = await fetchArticleDates(slug);
      
      if (published || updated) {
        const dateEl = createDateElement(published, updated);
        
        // Insert after the link (alongside bias info)
        mainLink.insertAdjacentElement('afterend', dateEl);
      }
    }
  }

  // Initial run
  await processArticleCards();

  // Watch for dynamic content changes (infinite scroll, etc.)
  let debounceTimer;
  const observer = new MutationObserver(() => {
    clearTimeout(debounceTimer);
    debounceTimer = setTimeout(processArticleCards, 500);
  });

  observer.observe(document.body, { 
    childList: true, 
    subtree: true 
  });

  // Also run on scroll (for lazy-loaded content)
  let scrollTimer;
  window.addEventListener('scroll', () => {
    clearTimeout(scrollTimer);
    scrollTimer = setTimeout(processArticleCards, 1000);
  });

})();

What this extension does:

Works on homepage (/) and /my feed
Fetches each article page in the background
Extracts both "Published" and "Updated" times
Displays them below each article card title
Caches results in localStorage to avoid refetching
Handles infinite scroll/dynamic content loading
Only shows "Updated" if it differs from "Published"
