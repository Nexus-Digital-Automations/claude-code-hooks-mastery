     ---
     Implementation Pattern

     Each script will be updated with:

     # New import
     import matplotlib
     matplotlib.use('Agg')  # Non-interactive backend
     import matplotlib.pyplot as plt

     # New argument in JSON metadata
     "chart_file": {
         "type": "string",
         "description": "Path to save visualization (PNG)",
         "required": false
     }

     # New function
     def generate_visualization(results, chart_file, analysis_type):
         """Generate and save visualization based on analysis type"""
         fig, axes = plt.subplots(...)
         # Plot logic based on analysis_type
         fig.savefig(chart_file, dpi=150, bbox_inches='tight')
         plt.close(fig)
         return chart_file

     # Update main_logic() to call visualization if chart_file provided
     if chart_file:
         generate_visualization(results, chart_file, analysis_type)
         results['chart_file'] = chart_file
